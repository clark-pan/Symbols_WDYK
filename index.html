ES6 symbols

what they are and how to use them

- Traceur and Babel allow us to write ES6 today - ES6 will make Javascript code look completely different - Symbols, more readable, more debuggable, possibly run faster

Basic usage

var nameSymbol = Symbol("name");

var person = {};
person[nameSymbol] = "Clark";
console.log(person[nameSymbol]); // Clark

person = {
	[nameSymbol] : "Clark"
; //Equivalent to above

console.log(typeof nameSymbol); // symbol

console.log(Symbol("name") === Symbol("name")); // false

console.log(person["name"]); // undefined
- Basic syntax and usage - get and set - Object literal usage - Lack of new operator, Symbol is a primitive - Descriptor has no relevance on the identity of the Symbol - Is actually using it as an object

Encapsulation without (ab)using closures

Hiding variables using closures


function Television(){
	var hasPower = false;

	this.turnOn = function(){
		console.log('Turned television on');
		hasPower = true;
	}

	this.switchChannel = function(){
		if(hasPower){
			console.log('Channel switched');
		} else {
			console.log('Nothing happened');
		}
	}
}

Hiding variables using symbols


var hasPowerKey = Symbol('hasPower');

export class Television {
	constructor() {
		this[hasPowerKey] = false;
	}
	turnOn() {
		console.log('Turned television on');
		this[hasPowerKey] = true;
	}
	switchChannel() {
		if(this[hasPowerKey]){
			console.log('Channel switched');
		} else {
			console.log('Nothing happened');
		}
	}
}
